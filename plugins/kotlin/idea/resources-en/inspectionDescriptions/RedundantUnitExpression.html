Reports redundant 'Unit' expression which can be omitted.

Examples:


    fun redundantUnitExp(b: Boolean): Unit = if (b) {
        class A
        Unit // <= is redundant here
    } else {
    }

    fun neededUnitExp(b: Boolean): Unit = if (b) {
        1
        Unit // <= while here it's needed
    } else {
    }
